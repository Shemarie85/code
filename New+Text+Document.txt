def display_menu():
    print('Country List program\n')
    print('COMMAND MENU')
    print('view - View a country')
    print('add - Add a country')
    print('del - Delete a country')
    print('exit - Exit program')

def view_country(country_dic):
    print('Country Keys: ')
    for key in country_dic:
        print(key)
    user_input = input('Enter key to display country: ')
    if user_input in country_dic:
         print('Country Name', country_dic[user_input])
    else:
        print('Invalid key')
        
def add_country(country_dic):
    key = input('Enter key for country: ')
    if key not in country_dic:
         country_name = input('Enter country name: ')
         country_dic[key] = country_name
         print('Country added')
    else:
        print('Key already exits')
        
def delete_country(country_dic):
    key = input('Enter key for country to delete: ')
    if key in country_dic:
         country_dic.pop(key)
         print('Country deleted')
    else:
        print('Key donot exits')
    

def start():
    country_dic = {}
    country_dic['U'] = 'USA'
    country_dic['C'] = 'China'
    country_dic['R'] = 'Russia'
    display_menu()
    while True:
        print()
        command = input('Command: ')
        command = command.lower()
        if command == 'view':
            view_country(country_dic)
        elif command == 'add':
             add_country(country_dic)   
        elif command == 'del':
            delete_country(country_dic)
        elif command == 'exit':
            break
        else:
            print('Not a valid command. Please try again.')
          
start()
